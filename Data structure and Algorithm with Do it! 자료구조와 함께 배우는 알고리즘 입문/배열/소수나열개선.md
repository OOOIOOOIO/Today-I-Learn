```java
package 배열;

public class 소수나열개선 {
	
	public static void main(String[] args) {
		int divCnt = 0; // 나눗셈 곱셈 횟수
		int primeNumCnt = 0; // 소수의 개수
		int[] primeNumber = new int[500]; // 소수를 저장할 배열 
		
		
		primeNumber[primeNumCnt++] = 2; // 0번째에 2
		primeNumber[primeNumCnt++] = 3; // 1번째에 3
		
		for(int i = 5; i <= 1000; i += 2) {
			boolean flag = false;
			
			for(int j = 1; primeNumber[j] * primeNumber[j] <= i; j++) {
				divCnt += 2; // []*[] , i % [] --> 2번 
				
				if(i % primeNumber[j] == 0) {
					flag = true;
					break;
				}
			}
			
			if(!flag) {
				primeNumber[primeNumCnt++] = i;
				divCnt++;
			}
		}
		
		for(int i = 0; i < primeNumCnt; i++) System.out.println(primeNumber[i]);
		
		
		System.out.println("곱셈과 나눗셈을 수행한 횟수 : " + divCnt);
	}
}
```

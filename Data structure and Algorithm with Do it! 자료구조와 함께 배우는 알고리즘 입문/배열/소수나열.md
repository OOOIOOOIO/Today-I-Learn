package 배열;

public class 소수나열 {// 1000 이하의 소수 찾기 
	public static void main(String[] args) {
		int divCnt = 0; // 나눗셈 횟수
		int primNumCnt =0; // 찾은 소수의 개수
		int[] primeNumber = new int[500]; // 소수를 저장하는 배열
		
		primeNumber[primNumCnt++] = 2; // [0]에 2(첫 소수)를 넣어주기
		
		for(int i = 3; i <= 5; i += 2) { // 홀수만 생각(짝수 != 소수)
			int j; // for문 밖에서도 쓰기 때문에 밖에서 선언.
			for(j = 1; j < primNumCnt; j++) {
				divCnt++;
				if(i % primeNumber[j] == 0) break; // 이미 찾은 소수로 나누어 떨어지면 소수가 아니다
				
			}
			if(primNumCnt == j) primeNumber[primNumCnt++] = i;
		}
		
		for(int i = 0; i < primNumCnt; i++) System.out.println(primeNumber[i]);
		
		System.out.println("나눗셈을 수행한 횟수 :" + divCnt);
	}
}
